  function getXmlHttp(){
    var xmlhttp;

      xmlhttp = new ActiveXObject("Msxml2.XMLHTTP");
    } catch (e) {

  function updateDataIDs(template, SrcDoc, indexes) {

    var elem;
    var srcelem;

    var resmore = false;

    if(indexes > 0) {
      do {
        more = updateDataIDs(template+'_'+index, SrcDoc, indexes-1);

        index++;
      } while(more);

  }
